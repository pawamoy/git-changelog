1# Why does this file exist, and why not put this in `__main__`? 

2# 

3# You might be tempted to import things from `__main__` later, 

4# but that will cause problems: the code will get executed twice: 

5# 

6# - When you run `python -m git_changelog` python will execute 

7# `__main__.py` as a script. That means there won't be any 

8# `git_changelog.__main__` in `sys.modules`. 

9# - When you import `__main__` it will get executed again (as a module) because 

10# there's no `git_changelog.__main__` in `sys.modules`. 

11

96def main(args: Optional[List[str]] = None) -> int: 

97 """ 

98 Run the main program. 

99 

100 This function is executed when you type `git-changelog` or `python -m git_changelog`. 

101 

102 Arguments: 

103 args: Arguments passed from the command line. 

104 

105 Returns: 

106 An exit code. 

107 """ 

108 parser = get_parser() 

109 opts = parser.parse_args(args=args) 

110 

111 # get template 

112 if opts.template.startswith("path:"): 

113 path = opts.template.replace("path:", "", 1) 

114 try: 

115 template = templates.get_custom_template(path) 

116 except TemplateNotFound: 

117 print(f"git-changelog: no such directory, or missing changelog.md: {path}", file=sys.stderr) 

118 return 1 

119 else: 

120 template = templates.get_template(opts.template) 

121 

122 # build data 

123 changelog = Changelog(opts.repository, style=opts.style) 

124 

125 # get rendered contents 

126 rendered = template.render(changelog=changelog) 

127 

128 # write result in specified output 

129 if opts.output is sys.stdout: 

130 sys.stdout.write(rendered) 

131 else: 

132 with open(opts.output, "w") as stream: 

133 stream.write(rendered) 

134 

135 return 0
